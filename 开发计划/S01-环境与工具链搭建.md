### 2.3 Mac 硬件评估脚本

文件：`scripts/check_hardware.py`

```python
import psutil
import platform
import subprocess

def check_hardware():
    """评估 Mac 硬件，决定 Hugo 构建策略"""
    mem = psutil.virtual_memory()
    mem_gb = mem.total / (1024**3)
    cpu_count = psutil.cpu_count(logical=True)

    # 评估 Hugo 构建策略
    if mem_gb >= 32:
        strategy = "full"  # 一次性构建
        concurrency = 32
    elif mem_gb >= 16:
        strategy = "batch"  # 分批构建
        concurrency = 16
    else:
        strategy = "batch_conservative"  # 保守分批
        concurrency = 8

    return {
        "memory_gb": round(mem_gb, 1),
        "cpu_count": cpu_count,
        "hugo_strategy": strategy,
        "asyncio_concurrency": concurrency,
        "macos_version": platform.mac_ver()[0],
    }

if __name__ == "__main__":
    import json
    print(json.dumps(check_hardware(), indent=2, ensure_ascii=False))
```

	python3 -m venv .venv
	. .venv/bin/activate && pip install -r requirements.txt
	cd site && hugo mod deps

	. .venv/bin/activate && python -m src.mock_server.app

hugo-server:
	cd site && hugo server -D --bind 0.0.0.0

pipeline:
	. .venv/bin/activate && python -m src.pipeline.orchestrator

build:
	cd site && hugo --minify --gc

test:
	. .venv/bin/activate && pytest tests/ -v --cov=src

	. .venv/bin/activate && bash scripts/local_e2e.sh

	. .venv/bin/activate && python scripts/perf_dashboard.py

## 六、风险提示

| 风险 | 等级 | 应对 |
|------|------|------|
| Hugo 版本低于 0.130 | 中 | `brew upgrade hugo`，确保 Extended 版本 |
| Python 版本低于 3.11 | 中 | `brew install python@3.11`，更新 symlink |
| Mac fd 上限未生效 | 低 | 重启终端，确认 `~/.zshrc` 配置 |
| 8GB Mac 内存不足 | 高 | 启用 `batch_conservative` 策略，分批构建 |
