```bash
#!/bin/bash
set -e

test -s /tmp/slice_translate.json && echo "  ✅ translate 通过"

# 4. illustrate
echo "[4/7] illustrate..."
python -m src.scripts.illustrate --date 07-04 --country CN > /tmp/slice_illustrate.json
test -s /tmp/slice_illustrate.json && echo "  ✅ illustrate 通过"

# 5. audit
echo "[5/7] audit..."
python -m src.scripts.audit --date 07-04 --country CN > /tmp/slice_audit.json
test -s /tmp/slice_audit.json && echo "  ✅ audit 通过"

# 6. publish
echo "[6/7] publish..."
python -m src.scripts.publish --date 07-04 --country CN
MARKDOWN_FILE="site/content/zh/CN/07-04.md"

test -f "$HTML_FILE" && echo "  ✅ Hugo 构建通过 ($HTML_FILE)"

echo "=== 切片验证完成 ==="
echo "浏览器访问: http://localhost:1313/zh/on-this-day/CN/07-04/"
```

### 2.2 切片验证测试

文件：`tests/integration/test_vertical_slice.py`

```python
import pytest
import asyncio
import subprocess
from pathlib import Path
from src.config import settings
from src.pipeline.orchestrator import run_pipeline

@pytest.mark.asyncio
async def test_vertical_slice_e2e():
    """垂直切片 E2E 测试"""
    # 1. 运行管道
    result = await run_pipeline("07-04", settings)
    assert result.published_files > 0

    # 2. 验证 Markdown 文件生成
    md_file = Path("site/content/zh/CN/07-04.md")
    assert md_file.exists()
    content = md_file.read_text(encoding="utf-8")
    assert "title:" in content
    assert "country_code:" in content

    # 3. Hugo 构建
    result = subprocess.run(
        ["hugo", "--quiet"],
        cwd="site",
        capture_output=True,
        text=True
    )
    assert result.returncode == 0

    # 4. 验证 HTML 生成
    html_file = Path("site/public/zh/on-this-day/CN/07-04/index.html")
    assert html_file.exists()

    # 5. 验证 HTML 内容
    html = html_file.read_text(encoding="utf-8")
    assert "<!DOCTYPE html>" in html
    assert "今天历史" in html or "Today in History" in html

def test_slice_performance():
    """切片性能验证"""
    import time
    start = time.perf_counter()
    subprocess.run(["bash", "scripts/vertical_slice.sh"], check=True)
    duration = time.perf_counter() - start
    assert duration < 300, f"切片耗时 {duration:.1f}s 超过 5min"
```
