### 2.4 生产验证脚本

文件：`scripts/production_verify.sh`

```bash
#!/bin/bash
set -e

SITE_URL="https://today-in-history.example.com"

echo "=== 生产环境验证 ==="

# 1. 首页可访问
echo "[1/8] 验证首页..."
STATUS=$(curl -s -o /dev/null -w "%{http_code}" "$SITE_URL/")
[ "$STATUS" = "200" ] && echo "  ✅ 首页 200" || { echo "  ❌ 首页 $STATUS"; exit 1; }

# 2. 详情页可访问
echo "[2/8] 验证详情页..."
STATUS=$(curl -s -o /dev/null -w "%{http_code}" "$SITE_URL/zh/on-this-day/CN/07-04/")
[ "$STATUS" = "200" ] && echo "  ✅ 详情页 200" || { echo "  ❌ 详情页 $STATUS"; exit 1; }

# 3. sitemap.xml 可访问
echo "[3/8] 验证 sitemap..."
STATUS=$(curl -s -o /dev/null -w "%{http_code}" "$SITE_URL/sitemap.xml")
[ "$STATUS" = "200" ] && echo "  ✅ sitemap 200" || { echo "  ❌ sitemap $STATUS"; exit 1; }

# 4. robots.txt 可访问
echo "[4/8] 验证 robots.txt..."
STATUS=$(curl -s -o /dev/null -w "%{http_code}" "$SITE_URL/robots.txt")
[ "$STATUS" = "200" ] && echo "  ✅ robots.txt 200" || { echo "  ❌ robots.txt $STATUS"; exit 1; }

# 5. CDN 命中
echo "[5/8] 验证 CDN..."
CF_STATUS=$(curl -s -I "$SITE_URL/" | grep -i "cf-cache-status" | awk '{print $2}' | tr -d '\r')
[ "$CF_STATUS" = "HIT" ] && echo "  ✅ CDN HIT" || echo "  ⚠️  CDN $CF_STATUS (首次访问可能 MISS)"

# 6. TTFB
echo "[6/8] 验证 TTFB..."
TTFB=$(curl -s -o /dev/null -w "%{time_starttransfer}" "$SITE_URL/")
echo "  TTFB: ${TTFB}s (目标 < 0.2s)"

# 7. SSL 证书
echo "[7/8] 验证 SSL..."
echo | openssl s_client -servername today-in-history.example.com -connect today-in-history.example.com:443 2>/dev/null | openssl x509 -noout -dates 2>/dev/null && echo "  ✅ SSL 有效" || echo "  ❌ SSL 错误"

# 8. Lighthouse
echo "[8/8] Lighthouse 性能测试..."
npx lighthouse "$SITE_URL/" \
  --output json \
  --output-path 09_报告/lighthouse_prod.json \
  --throttling-method=simulate \
  --only-categories=performance \
  --chrome-flags="--headless" 2>/dev/null

PERF=$(python -c "import json; d=json.load(open('09_报告/lighthouse_prod.json')); print(d['categories']['performance']['score'])")
PERF_PCT=$(python -c "print(int(${PERF} * 100))")
echo "  Performance: ${PERF_PCT}/100 (目标 > 90)"

echo ""
echo "=== 验证完成 ==="
```
