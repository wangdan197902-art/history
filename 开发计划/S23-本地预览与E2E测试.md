### 2.2 E2E 测试脚本

文件：`tests/e2e/test_full_flow.py`

```python
"""E2E 测试: 完整流程"""
import subprocess
import time
import requests
import pytest
from pathlib import Path

def test_mock_server_health():
    """测试 Mock Server 健康"""
    resp = requests.get("http://127.0.0.1:8765/health", timeout=5)
    assert resp.status_code == 200
    assert resp.json()["status"] == "ok"

def test_wikipedia_mock_response():
    """测试 Wikipedia Mock 响应"""
    resp = requests.get(
        "http://127.0.0.1:8765/wikipedia/onthisday/events/07/04",
        params={"country": "CN"},
        timeout=10
    )
    assert resp.status_code == 200
    data = resp.json()
    assert "events" in data

def test_pipeline_single_day():
    """测试单日管道"""
    result = subprocess.run(
        ["python", "-m", "src.pipeline.orchestrator", "07-04"],
        capture_output=True,
        text=True,
        timeout=600
    )
    assert result.returncode == 0

def test_hugo_build():
    """测试 Hugo 构建"""
    result = subprocess.run(
        ["bash", "scripts/hugo_build.sh"],
        capture_output=True,
        text=True,
        timeout=900
    )
    assert result.returncode == 0
    assert Path("site/public/index.html").exists()

def test_hugo_server_accessibility():
    """测试 Hugo server 可访问性"""
    # 启动 Hugo server
    proc = subprocess.Popen(
        ["hugo", "server", "--port", "1314"],
        cwd="site",
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL
    )
    try:
        time.sleep(3)
        resp = requests.get("http://localhost:1314/", timeout=5)
        assert resp.status_code == 200
        assert "今天历史" in resp.text or "Today in History" in resp.text
    finally:
        proc.terminate()
        proc.wait()

def test_specific_page_accessible():
    """测试特定页面可访问"""
    proc = subprocess.Popen(
        ["hugo", "server", "--port", "1315"],
        cwd="site",
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL
    )
    try:
        time.sleep(3)
        resp = requests.get(
            "http://localhost:1315/zh/on-this-day/CN/07-04/",
            timeout=5
        )
        assert resp.status_code == 200
        assert "7月4日" in resp.text or "07-04" in resp.text
    finally:
        proc.terminate()
        proc.wait()

def test_html_size_budget():
    """测试 HTML 体积预算"""
    html_files = list(Path("site/public").rglob("*.html"))
    assert len(html_files) >= 100000

    oversized = [f for f in html_files if f.stat().st_size > 50 * 1024]
    assert len(oversized) < 100, f"超大页面数: {len(oversized)}"
```

### 2.3 Lighthouse 性能测试

文件：`scripts/lighthouse_test.sh`

```bash
#!/bin/bash
set -e

echo "=== Lighthouse 性能测试 ==="

# 启动 Hugo server
cd site && hugo server --port 1316 &
HUGO_PID=$!
trap "kill $HUGO_PID" EXIT
sleep 3

# 运行 Lighthouse
npx lighthouse http://localhost:1316/ \
  --output html \
  --output-path ../09_报告/lighthouse_home.html \
  --throttling-method=simulate \
  --only-categories=performance,accessibility,best-practices,seo \
  --chrome-flags="--headless"

npx lighthouse http://localhost:1316/zh/on-this-day/CN/07-04/ \
  --output html \
  --output-path ../09_报告/lighthouse_day.html \
  --throttling-method=simulate \
  --only-categories=performance,accessibility,best-practices,seo \
  --chrome-flags="--headless"

echo "=== Lighthouse 报告已生成 ==="
echo "  - 首页: 09_报告/lighthouse_home.html"
echo "  - 详情页: 09_报告/lighthouse_day.html"
```
