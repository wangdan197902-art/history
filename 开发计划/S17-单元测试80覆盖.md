### 2.3 Property-Based Testing（增强包 3）

文件：`tests/strategies.py`

```python
from hypothesis import strategies as st
from src.models.event import Event

date_strategy = st.tuples(
    st.integers(min_value=1, max_value=12).map(lambda x: f"{x:02d}"),
    st.integers(min_value=1, max_value=28).map(lambda x: f"{x:02d}"),
).map(lambda t: f"{t[0]}-{t[1]}")

event_strategy = st.builds(
    Event,
    id=st.text(min_size=1, max_size=50),
    date=st.just("2026-07-04"),
    year=st.integers(min_value=1900, max_value=2026),
    title=st.text(min_size=1, max_size=200),
    description=st.text(min_size=1, max_size=1000),
    wikipedia_url=st.urls(),
)
```

### 2.4 集成测试

文件：`tests/integration/test_pipeline.py`

```python
import pytest
import asyncio
from src.config import settings
from src.pipeline.orchestrator import run_pipeline

@pytest.mark.asyncio
async def test_single_day_pipeline():
    """单日管道集成测试"""
    result = await run_pipeline("07-04", settings)
    assert result.published_files > 0
    assert result.pass_rate > 0.8

@pytest.mark.asyncio
async def test_5_sample_dates():
    """5 日期样本测试"""
    for d in ["01-01", "02-29", "07-04", "10-01", "12-25"]:
        result = await run_pipeline(d, settings)
        assert result.total_events > 0
```
