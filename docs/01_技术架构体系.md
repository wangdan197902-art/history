### 2.2 内容目录组织规则

按 `content/{lang}/{country}/{MM-DD}.md` 组织，URL 形如 `/en/on-this-day/us/07-04/`。

```
content/en/us/
├── _index.md             # 美国历史首页
├── 01-01.md              # 1月1日
├── 01-02.md
├── ...
└── 12-31.md

content/zh/cn/
├── _index.md
├── 01-01.md
└── ...
```

每个 Markdown 文件包含 Front Matter + 正文，正文通过 `{{< events >}}` 短代码渲染 JSON 数据。

### 2.3 命名约定

| 对象 | 规则 | 示例 |
|------|------|------|
| 内容文件 | `{MM-DD}.md` | `07-04.md` |
| 事件数据 | `{MM-DD}.json` | `07-04.json` |
| 图片 | `{country}-{MM-DD}-{seq}.webp` | `us-07-04-01.webp` |
| 地区代码 | ISO 3166-1 alpha-2 小写 | `us`, `cn`, `jp` |
| 语言代码 | ISO 639-1 小写 | `en`, `zh`, `es` |
| AI 日志 | `{stage}-{YYYYMMDD}.log` | `fetch-20260718.log` |

## 4. 数据模型设计

### 4.1 历史事件 JSON Schema

文件路径：`data/events/{MM-DD}.json`

```json
{
  "$schema": "https://json-schema.org/draft/2020-12/schema",
  "type": "object",
  "required": ["date", "events"],
  "properties": {
    "date": {
      "type": "object",
      "required": ["month", "day"],
      "properties": {
        "month": { "type": "integer", "minimum": 1, "maximum": 12 },
        "day":   { "type": "integer", "minimum": 1, "maximum": 31 },
        "iso":   { "type": "string", "pattern": "^\\d{2}-\\d{2}$" }
      }
    },
    "events": {
      "type": "array",
      "items": {
        "type": "object",
        "required": ["id", "year", "title", "description", "type", "countries", "sources", "neutrality_score"],
        "properties": {
          "id": {
            "type": "string",
            "description": "事件唯一 ID，格式 {MMDD}-{slug}",
            "pattern": "^\\d{4}-[a-z0-9-]+$"
          },
          "year": {
            "type": "integer",
```

### 5.3 管道编排脚本 `scripts/pipeline.py`

```python
#!/usr/bin/env python3
"""
地区化今天历史档案站 — 每日 AI 生产管道编排
运行环境：GitHub Actions (daily-history.yml)
依赖：python-dotenv, requests, anthropic, openai, google-generativeai
"""
import os
import sys
import json
import logging
import subprocess
from datetime import datetime, timezone, timedelta
from pathlib import Path

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(name)s: %(message)s"
)
log = logging.getLogger("pipeline")

REPO_ROOT = Path(__file__).parent.parent
DATA_DIR = REPO_ROOT / "data" / "events"
CONTENT_DIR = REPO_ROOT / "content"
LOG_DIR = REPO_ROOT / "logs" / "pipeline"
LOG_DIR.mkdir(parents=True, exist_ok=True)


def step_fetch_events(target_date):
    """Stage 1: 抓取当日全球事件"""
    log.info("Stage 1: fetch_events(%s)", target_date)
    result = subprocess.run(
        ["python3", "scripts/fetch_events.py", "--date", target_date],
        capture_output=True, text=True, check=True
    )
    log.info("fetch_events stdout: %s", result.stdout[-500:])
    return DATA_DIR / f"{target_date[5:]}.json"  # MM-DD.json


def step_regionalize(events_file):
    """Stage 2: Claude 地区化重写"""
    log.info("Stage 2: regionalize(%s)", events_file)
    subprocess.run(
        ["python3", "scripts/regionalize.py", "--input", str(events_file)],
        check=True
    )


def step_translate(events_file):
    """Stage 3: GPT-4o 多语种翻译"""
    log.info("Stage 3: translate(%s)", events_file)
    subprocess.run(
        ["python3", "scripts/translate.py", "--input", str(events_file)],
        check=True
    )


def step_illustrate(events_file):
    """Stage 4: Gemini 配图 + Wikimedia 检索"""
    log.info("Stage 4: illustrate(%s)", events_file)
    subprocess.run(
        ["python3", "scripts/illustrate.py", "--input", str(events_file)],
        check=True
    )


def step_audit(events_file):
    """Stage 5: GPT-4o AI 审核"""
    log.info("Stage 5: audit(%s)", events_file)
    subprocess.run(
        ["python3", "scripts/audit.py", "--input", str(events_file)],
        check=True
    )


def step_publish(events_file):
    """Stage 6-7: 生成 Markdown + git commit"""
    log.info("Stage 6-7: publish(%s)", events_file)
    subprocess.run(
        ["python3", "scripts/publish.py", "--input", str(events_file)],
        check=True
    )


def run_pipeline(target_date: str):
    """运行完整管道"""
    log_file = LOG_DIR / f"pipeline-{target_date}.log"
    file_handler = logging.FileHandler(log_file)
    file_handler.setFormatter(logging.Formatter("%(asctime)s [%(levelname)s] %(message)s"))
    logging.getLogger().addHandler(file_handler)

    log.info("===== Pipeline start: %s =====", target_date)
    try:
        events_file = step_fetch_events(target_date)
        step_regionalize(events_file)
        step_translate(events_file)
        step_illustrate(events_file)
        step_audit(events_file)
        step_publish(events_file)
        log.info("===== Pipeline success =====")
        return 0
    except subprocess.CalledProcessError as e:
        log.exception("Stage failed: %s", e)
        # 通知人工介入
        notify_failure(target_date, str(e))
        return 1


def notify_failure(date, error):
    """失败时通过 GitHub Issue 通知"""
    # 详见 05_部署与运维方案.md
    pass


if __name__ == "__main__":
    # 默认抓取次日内容（UTC 01:00 调度，已进入次日）
    tomorrow = (datetime.now(timezone.utc) + timedelta(days=1)).strftime("%Y-%m-%d")
    target = sys.argv[1] if len(sys.argv) > 1 else tomorrow
    sys.exit(run_pipeline(target))
```

### 6.1 GitHub Actions 完整工作流

**文件：`.github/workflows/daily-history.yml`**

```yaml
name: Daily History Update

on:
  schedule:
    - cron: '0 1 * * *'  # UTC 01:00 每日触发
  workflow_dispatch:
    inputs:
      date:
        description: 'Target date (YYYY-MM-DD)'
        required: false

permissions:
  contents: write
  deployments: write

env:
  HUGO_VERSION: '0.130.0'
  PYTHON_VERSION: '3.11'

concurrency:
  group: daily-history
  cancel-in-progress: false

jobs:
  generate-and-deploy:
    runs-on: ubuntu-latest
    timeout-minutes: 60
    steps:
      - name: Checkout
        uses: actions/checkout@v4
        with:
          fetch-depth: 1

      - name: Setup Hugo
        uses: peaceiris/actions-hugo@v3
        with:
          hugo-version: ${{ env.HUGO_VERSION }}
          extended: true

      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: ${{ env.PYTHON_VERSION }}
          cache: 'pip'

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r scripts/requirements.txt

      - name: Determine target date
        id: date
        run: |
          if [ -n "${{ github.event.inputs.date }}" ]; then
            echo "target=${{ github.event.inputs.date }}" >> $GITHUB_OUTPUT
          else
            # UTC 次日（管道提前生成）
            TARGET=$(python -c "from datetime import datetime,timezone,timedelta; print((datetime.now(timezone.utc)+timedelta(days=1)).strftime('%Y-%m-%d'))")
            echo "target=$TARGET" >> $GITHUB_OUTPUT
          fi

      - name: Run AI pipeline
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
          OPENAI_API_KEY: ${{ secrets.OPENAI_API_KEY }}
          GEMINI_API_KEY: ${{ secrets.GEMINI_API_KEY }}
          WIKI_USER_AGENT: ${{ secrets.WIKI_USER_AGENT }}
        run: |
          python3 scripts/pipeline.py ${{ steps.date.outputs.target }}

      - name: Build Hugo
        run: |
          hugo --minify --gc --enableGitInfo \
            --baseURL https://today-in-history.example.com/

      - name: Commit new content
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "41898282+github-actions[bot]@users.noreply.github.com"
          git add content/ data/ assets/images/
          git diff --staged --quiet || git commit -m "chore(content): daily update ${{ steps.date.outputs.target }}"
          git push

      - name: Deploy to Cloudflare Pages
        uses: cloudflare/pages-action@v1
        with:
          apiToken: ${{ secrets.CF_API_TOKEN }}
          accountId: ${{ secrets.CF_ACCOUNT_ID }}
          projectName: today-in-history
          directory: public
          gitHubComments: true

      - name: Submit sitemap to search engines
        env:
          GSC_API_KEY: ${{ secrets.GSC_API_KEY }}
          GSC_SITE_URL: ${{ secrets.GSC_SITE_URL }}
        run: python3 scripts/submit_sitemap.py

      - name: Send daily email (Buttondown)
        env:
          BUTTONDOWN_API_KEY: ${{ secrets.BUTTONDOWN_API_KEY }}
        run: python3 scripts/send_daily_email.py --date ${{ steps.date.outputs.target }}

      - name: Notify on failure
        if: failure()
        uses: actions/github-script@v7
        with:
          script: |
            github.rest.issues.create({
              owner: context.repo.owner,
              repo: context.repo.repo,
              title: `Pipeline failed: ${{ steps.date.outputs.target }}`,
              body: `Run: ${context.runId}\nDate: ${{ steps.date.outputs.target }}\nLogs: ${context.serverUrl}/${context.repo.owner}/${context.repo.repo}/actions/runs/${context.runId}`,
              labels: ['pipeline-failure']
            });
```

### 7.1 URL 与 Canonical 策略

**URL 结构：** `/{lang}/on-this-day/{country}/{MM-DD}/`

**Canonical 处理（关键）：**

历史事件的日期是"年-月-日"，但 URL 只有"月-日"。多年事件汇聚到同一 URL，需要：

```html
<!-- 07-04 页面 canonical 指向自身（不指向某一年） -->
<link rel="canonical" href="https://today-in-history.example.com/en/on-this-day/us/07-04/" />

<!-- 多语种 hreflang 互链 -->
<link rel="alternate" hreflang="en" href="https://.../en/on-this-day/us/07-04/" />
<link rel="alternate" hreflang="zh" href="https://.../zh/on-this-day/us/07-04/" />
<link rel="alternate" hreflang="x-default" href="https://.../en/on-this-day/us/07-04/" />
```

### 8.4 Cloudflare 边缘缓存策略

**Cloudflare Workers 路由规则**（可选增强）：

```javascript
// workers/cache-rules.js
export default {
  async fetch(request, env) {
    const url = new URL(request.url);

    // 历史日期页：边缘缓存 4 小时（每月同日不变化）
    if (url.pathname.match(/\/on-this-day\/[a-z]{2}\/[a-z]{2}\/\d{2}-\d{2}\//)) {
      const response = await fetch(request);
      response.headers.set('Cache-Control', 's-maxage=14400, stale-while-revalidate=86400');
      return response;
    }

    // 静态资源：缓存 1 年
    if (url.pathname.match(/\.(webp|jpg|png|css|js|woff2)$/)) {
      const response = await fetch(request);
      response.headers.set('Cache-Control', 'public, max-age=31536000, immutable');
      return response;
    }

    return fetch(request);
  }
};
```

```python
        if r.status_code != 200:

        today = datetime.now(timezone.utc).strftime("%B %-d")
        if today not in r.text:
            return {"status": "warn", "msg": f"today's date '{today}' not on homepage"}
        return {"status": "ok"}
    except Exception as e:
        return {"status": "fail", "msg": str(e)}

    # 调用 GSC API（详见 04_SEO流量与运营变现.md）
    pass


def check_ai_cost():
    """检查 AI API 累计成本"""
    # 各平台 API 调用，超阈值告警
    pass


def alert(message):
    """发送告警"""
    if not ALERT_WEBHOOK:
        log.warning("ALERT_WEBHOOK not set, skipping: %s", message)
        return
    requests.post(ALERT_WEBHOOK, json={"text": f"🚨 Today-in-History: {message}"})


def main():

        except Exception as e:
            report[name] = {"status": "error", "msg": str(e)}
            alert(f"{name} crashed: {e}")

    out = Path("logs/monitoring") / f"check-{datetime.now(timezone.utc).strftime('%Y%m%d')}.json"
    out.parent.mkdir(parents=True, exist_ok=True)
```
