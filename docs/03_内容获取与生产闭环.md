### 1.2 关键节点

| 节点 | 触发条件 | 输入 | 输出 | 责任方 |

### 1.3 运行节奏

- **每日**：UTC 01:00 启动 → UTC 03:00 发布 → UTC 06:00 邮件推送
- **每周**：周一编辑抽检 5% 主要国家 + 优化 Prompt
- **每月**：月初生成月度报告 + Prompt 模板迭代
- **年度**：1 月全量刷新 366 天 × 30 地区 + AI 模型升级评估

### 2.3 完整 `fetch_events.py` 脚本

**文件：`scripts/fetch_events.py`**

```python
#!/usr/bin/env python3
"""
历史事件采集 — Wikipedia OnThisDay API + 多语种扩展 + 30地区过滤
依赖：requests, python-dotenv
运行：python3 scripts/fetch_events.py --date 2026-07-04
输出：data/events/07-04.json
"""
import os
import sys
import json
import logging
import argparse
from datetime import datetime
from pathlib import Path
import requests
from dotenv import load_dotenv

load_dotenv()

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(message)s"
)
log = logging.getLogger("fetch")

REPO_ROOT = Path(__file__).parent.parent
DATA_DIR = REPO_ROOT / "data" / "events"
DATA_DIR.mkdir(parents=True, exist_ok=True)

WIKI_USER_AGENT = os.getenv(
    "WIKI_USER_AGENT",
    "TodayInHistoryArchive/1.0 (contact@example.com)"
)
HEADERS = {"User-Agent": WIKI_USER_AGENT}

# 30 地区 ISO 代码与 Wikipedia 语种对应
COUNTRY_LANG_MAP = {
    "us": "en", "gb": "en", "ca": "en", "au": "en", "nz": "en", "in": "en", "za": "en",
    "de": "de", "at": "de",
    "fr": "fr",
    "es": "es", "mx": "es", "ar": "es",
    "it": "it",
    "pt": "pt", "br": "pt",
    "ru": "ru",
    "cn": "zh",
    "jp": "ja",
    "kr": "ko",
    "tr": "tr",
    "gr": "el",
    "pl": "pl",
    "ua": "uk",
    "se": "sv",
    "no": "no",
    "nl": "nl", "be": "nl",
    "ie": "ga",
    "eg": "ar",
}

# 9 个 Wikipedia 语种端点（去重）
WIKI_LANGS = sorted(set(COUNTRY_LANG_MAP.values()))

WIKI_ENDPOINT_TEMPLATE = "https://{lang}.wikipedia.org/api/rest_v1/feed/onthisday/events/{month:02d}/{day:02d}"


def fetch_wiki_events(lang: str, month: int, day: int) -> list:
    """从某个语种 Wikipedia 拉取当日事件"""
    url = WIKI_ENDPOINT_TEMPLATE.format(lang=lang, month=month, day=day)
    log.info("Fetching %s", url)
    try:
        resp = requests.get(url, headers=HEADERS, timeout=30)
        resp.raise_for_status()
        data = resp.json()
        events = data.get("events", [])
        log.info("  %s: %d events", lang, len(events))
        return events
    except requests.RequestException as e:
        log.error("Failed to fetch %s: %s", lang, e)
        return []


def normalize_event(raw_event: dict, source_lang: str) -> dict:
    """将 Wikipedia 事件标准化为内部 schema"""
    pages = raw_event.get("pages", [])
    primary_page = pages[0] if pages else {}
    thumbnail = primary_page.get("thumbnail", {})

    return {
        "id": f"{raw_event.get('year', '????')}-{slugify(primary_page.get('title', raw_event.get('text', '')[:30]))}"[:60],
        "year": raw_event.get("year"),
        "year_precision": "year",
        "title": primary_page.get("title") or raw_event.get("text", "")[:120],
        "description": raw_event.get("text", ""),
        "type": classify_event(primary_page.get("description", "") + " " + raw_event.get("text", "")),
        "countries": [],  # Stage 2 由 Claude 填充
        "people": extract_people(pages),
        "places": extract_places(primary_page),
        "image": {
            "url": thumbnail.get("url"),
            "alt": primary_page.get("title", ""),
            "caption": primary_page.get("description", ""),
            "source": f"Wikipedia ({source_lang})",
            "license": "cc-by-sa",
            "credit": primary_page.get("title", "")
        } if thumbnail else None,
        "sources": [
            {
                "name": f"Wikipedia ({source_lang}): {primary_page.get('title', '')}",
                "url": primary_page.get("content_urls", {}).get("desktop", {}).get("page", ""),
                "retrieved": datetime.utcnow().strftime("%Y-%m-%d")
            }
        ] if primary_page else [],
        "neutrality_score": None,  # Stage 5 由 AI 评估
        "sensitivity_flags": ["none"],  # Stage 5 由 AI 重新评估
        "locales": {},  # Stage 3 由 GPT-4o 翻译填充
        "_source_lang": source_lang,
        "_wiki_pages": [p.get("title") for p in pages]
    }


def slugify(text: str) -> str:
    """生成 URL 友好的 slug"""
    import re
    text = text.lower().strip()
    text = re.sub(r"[^\w\s-]", "", text)
    text = re.sub(r"[\s_]+", "-", text)
    return text[:50].strip("-")


def classify_event(text: str) -> str:
    """初步事件分类（Stage 5 AI 会重新精分）"""
    text_lower = text.lower()
    keyword_map = {
        "military": ["war", "battle", "invasion", "army", "soldier", "military", "attack", "siege", "troops", "navy"],
        "political": ["election", "government", "president", "parliament", "congress", "king", "queen", "minister", "revolution", "independence"],
        "scientific": ["discovery", "invention", "scientist", "physics", "chemistry", "medicine", "patent", "nobel"],
        "cultural": ["art", "music", "literature", "book", "film", "painting", "opera", "theatre", "festival"],
        "sporting": ["olympic", "championship", "cup", "tournament", "athlete", "record", "soccer", "football", "baseball"],
        "disaster": ["earthquake", "tsunami", "eruption", "hurricane", "crash", "explosion", "fire", "flood", "accident"],
        "biographical": ["born", "died", "death", "birth", "assassinated", "killed"],
        "economic": ["stock", "market", "bank", "currency", "economic", "trade", "tariff"],
        "religious": ["pope", "church", "temple", "religious", "catholic", "protestant", "islam", "buddhist"],
        "exploration": ["expedition", "voyage", "discovered", "explorer", "space", "moon", "mars"],
        "legal": ["court", "law", "supreme court", "ruling", "verdict", "constitution", "amendment"],
        "social": ["protest", "march", "movement", "demonstration", "riot", "civil rights"],
    }
    for category, keywords in keyword_map.items():
        if any(kw in text_lower for kw in keywords):
            return category
    return "social"  # 默认分类


def extract_people(pages: list) -> list:
    """从 Wikipedia pages 提取相关人物"""
    people = []
    for page in pages[:5]:
        if page.get("type") in ("person", "standard"):
            people.append({
                "name": page.get("title"),
                "role": page.get("description", ""),
                "wiki_url": page.get("content_urls", {}).get("desktop", {}).get("page", "")
            })
    return people


def extract_places(page: dict) -> list:
    """从 Wikipedia page 提取地点（简化版）"""
    # 完整实现需要调用 Wikipedia coordinates API
    title = page.get("title", "")
    if title:
        return [{"name": title, "lat": None, "lon": None, "country": None}]
    return []


def fetch_all_langs(month: int, day: int) -> dict:
    """从所有 9 个语种 Wikipedia 拉取事件并合并去重"""
    all_events = []
    seen_titles = set()

    for lang in WIKI_LANGS:
        events = fetch_wiki_events(lang, month, day)
        for evt in events:
            normalized = normalize_event(evt, lang)
            # 去重：以标题（小写）为 key
            key = normalized["title"].lower()
            if key not in seen_titles:
                seen_titles.add(key)
                all_events.append(normalized)

    log.info("Total unique events: %d", len(all_events))
    return {
        "date": {"month": month, "day": day, "iso": f"{month:02d}-{day:02d}"},
        "events": all_events,
        "metadata": {
            "fetched_at": datetime.utcnow().isoformat() + "Z",
            "sources": [f"Wikipedia ({lang})" for lang in WIKI_LANGS],
            "raw_count": sum(len(fetch_wiki_events(l, month, day)) for l in WIKI_LANGS),
            "deduped_count": len(all_events)
        }
    }


def main():
    parser = argparse.ArgumentParser(description="Fetch historical events for a date")
    parser.add_argument("--date", required=True, help="Target date YYYY-MM-DD")
    parser.add_argument("--output", help="Output file (default: data/events/MM-DD.json)")
    args = parser.parse_args()

    target = datetime.strptime(args.date, "%Y-%m-%d")
    month, day = target.month, target.day

    log.info("===== Fetch events for %04d-%02d-%02d =====", target.year, month, day)

    data = fetch_all_langs(month, day)

    output_file = Path(args.output) if args.output else DATA_DIR / f"{month:02d}-{day:02d}.json"
    output_file.parent.mkdir(parents=True, exist_ok=True)
    output_file.write_text(json.dumps(data, indent=2, ensure_ascii=False))
    log.info("Saved: %s (%d events)", output_file, len(data["events"]))


if __name__ == "__main__":
    main()
```

## 3. 地区化适配模块（Claude）

### 3.1 模块职责

- 输入：Stage 1 采集的全球事件池（200-500 事件/日）
- 输出：30 地区筛选+重写 JSON（每地区 5-15 事件）
- AI：Claude 3.5 Sonnet（长上下文 + 中立性写作最佳）

### 3.2 完整 `regionalize.py` 脚本

```python
#!/usr/bin/env python3
"""
地区化适配 — Claude 3.5 Sonnet 为 30 地区筛选并重写事件
依赖：anthropic, python-dotenv
运行：python3 scripts/regionalize.py --input data/events/07-04.json
"""
import os
import sys
import json
import logging
import argparse
from pathlib import Path
from anthropic import Anthropic
from dotenv import load_dotenv

load_dotenv()

logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s")
log = logging.getLogger("regionalize")

client = Anthropic(api_key=os.getenv("ANTHROPIC_API_KEY"))
MODEL = "claude-3-5-sonnet-20241022"

COUNTRIES = [
    ("us", "United States"), ("gb", "United Kingdom"), ("de", "Germany"),
    ("fr", "France"), ("es", "Spain"), ("it", "Italy"), ("ru", "Russia"),
    ("cn", "China"), ("jp", "Japan"), ("kr", "South Korea"), ("in", "India"),
    ("br", "Brazil"), ("mx", "Mexico"), ("ar", "Argentina"), ("ca", "Canada"),
    ("au", "Australia"), ("nz", "New Zealand"), ("za", "South Africa"),
    ("eg", "Egypt"), ("tr", "Turkey"), ("gr", "Greece"), ("pl", "Poland"),
    ("ua", "Ukraine"), ("se", "Sweden"), ("no", "Norway"), ("nl", "Netherlands"),
    ("be", "Belgium"), ("pt", "Portugal"), ("ie", "Ireland"), ("at", "Austria"),
]

# 敏感话题关键词（必须人工审核）
SENSITIVE_KEYWORDS = {
    "de": ["nazi", "hitler", "holocaust", "ss ", "gestapo", "third reich"],
    "cn": ["tiananmen", "cultural revolution", "great leap", "falun gong"],
    "tr": ["armenian", "genocide", "kurds", "kurdish"],
    "ru": ["putin", "ukraine war", "crimea", "annexation"],
    "ua": ["putin", "russian invasion", "donetsk", "luhansk"],
    "jp": ["nanjing", "comfort women", "yasukuni", "war crimes"],
    "kr": ["kim jong", "korean war", "comfort women"],
}


def build_prompt(events_data: dict, country_code: str, country_name: str) -> str:
    """构建 Claude 地区化 Prompt"""
    events_text = json.dumps(events_data["events"], ensure_ascii=False, indent=2)

    sensitive_keywords = SENSITIVE_KEYWORDS.get(country_code, [])

    return f"""You are a senior history editor for a regional "Today in History" archive.

# Task
Filter and rewrite historical events for {country_name} ({country_code}) for the date {events_data['date']['month']:02d}-{events_data['date']['day']:02d}.

# Input
The following is the global event pool (JSON):
```json
{events_text}
```

# Filtering Criteria (apply strictly)
1. Events that occurred within the territory of {country_name}
2. Events involving citizens/government of {country_name}
3. Events with significant impact on {country_name}
4. Aim for 5-15 events per country (skip if fewer than 3 qualify)

# Rewriting Rules (CRITICAL — Neutrality Principle)
1. Rewrite each event in 50-150 words narrative style (NOT Wikipedia verbatim)
2. Use NEUTRAL, FACTUAL tone — no emotional adjectives, no judgmental language
3. Multiple perspectives when event involves international conflict
4. Avoid loaded terms: "liberation" vs "invasion", "terrorist" vs "rebel" — use neutral descriptors
5. Cite sources: keep Wikipedia reference, add 1+ independent source if possible
6. Preserve original year, places, people
7. Sensitive topics for {country_code}: {sensitive_keywords or 'none specified'}
   - For sensitive events, provide multiple viewpoints where applicable
   - Use academic terminology, not partisan framing
8. Output in English (Stage 3 will translate to other languages)

# Output Format (JSON, no markdown wrapping)
{{
  "country_code": "{country_code}",
  "country_name": "{country_name}",
  "events_count": <int>,
  "events": [
    {{
      "id": "<original_id>",
      "year": <int>,
      "title": "<rewritten English title, max 100 chars>",
      "description": "<rewritten English narrative, 50-150 words>",
      "type": "<one of: political, military, scientific, cultural, sporting, disaster, biographical, economic, religious, exploration, legal, social>",
      "relevance": "primary|secondary|mentioned",
      "local_angle": "<1 sentence local perspective for {country_name} residents, max 200 chars>",
      "people": [{{"name": "...", "role": "..."}}],
      "places": [{{"name": "...", "country": "{country_code}"}}],
      "sources": [{{"name": "...", "url": "..."}}],
      "neutrality_score": <0.0-1.0, self-evaluation>,
      "sensitivity_flags": ["<flag>", ...]
    }}
  ]
}}

# Constraints
- Return ONLY valid JSON (no ```json markdown blocks)
- If fewer than 3 events qualify, return events_count: 0 and events: []
- Preserve all original source URLs
"""


def call_claude(prompt: str, max_tokens: int = 8000) -> dict:
    """调用 Claude API"""
    response = client.messages.create(
        model=MODEL,
        max_tokens=max_tokens,
        temperature=0.3,  # 低温度保证事实准确性
        messages=[{"role": "user", "content": prompt}]
    )
    text = response.content[0].text.strip()
    # 移除可能的 markdown 包裹
    if text.startswith("```"):
        text = text.split("\n", 1)[1].rsplit("```", 1)[0].strip()
    return json.loads(text)


def regionalize_for_country(events_data: dict, country_code: str, country_name: str) -> dict:
    """为单个地区生成地区化内容"""
    log.info("Regionalizing for %s (%s)", country_name, country_code)
    prompt = build_prompt(events_data, country_code, country_name)
    try:
        result = call_claude(prompt)
        log.info("  → %d events (avg neutrality: %.2f)",
                 result.get("events_count", 0),
                 sum(e.get("neutrality_score", 0) for e in result.get("events", [])) / max(len(result.get("events", [])), 1))
        return result
    except Exception as e:
        log.error("Failed for %s: %s", country_code, e)
        return {"country_code": country_code, "country_name": country_name,
                "events_count": 0, "events": [], "error": str(e)}


def main():
    parser = argparse.ArgumentParser()

### 4.1 模块职责

- 输出：10 语种 × 30 地区 = 300 个本地化版本

### 4.2 完整 `translate.py` 脚本

```python
#!/usr/bin/env python3
"""
多语言翻译 — GPT-4o 将英语版本翻译为 10 语种
依赖：openai, python-dotenv
运行：python3 scripts/translate.py --input data/events/07-04_regionalized.json
"""
import os
import sys
import json
import logging
import argparse
from pathlib import Path
from openai import OpenAI
from dotenv import load_dotenv

load_dotenv()

logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s")
log = logging.getLogger("translate")

client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))
MODEL = "gpt-4o"

# 10 目标语种
TARGET_LANGS = {
    "en": "English",
    "zh": "Chinese (Simplified, 简体中文)",
    "es": "Spanish (Español)",
    "fr": "French (Français)",
    "de": "German (Deutsch)",
    "it": "Italian (Italiano)",
    "pt": "Portuguese (Português brasileiro)",
    "ru": "Russian (Русский)",
    "ja": "Japanese (日本語)",
    "ko": "Korean (한국어)",
}


def build_translation_prompt(content: dict, target_lang_code: str, target_lang_name: str) -> str:
    """构建翻译 Prompt"""
    return f"""You are a professional translator specializing in historical content.

# Task
Translate the following English historical content into {target_lang_name}.

# Source Content (JSON)
```json
{json.dumps(content, ensure_ascii=False, indent=2)}
```

# Translation Rules
1. Translate `title` and `description` and `local_angle` fields into {target_lang_name}
2. Translate `country_name` if present
3. Proper nouns (people names, place names): use the LOCAL convention in {target_lang_name}
   - Example: "Tokyo" → "東京" (Japanese), "东京" (Chinese), "Tokio" (German)
   - Example: "Napoleon Bonaparte" → "拿破仑·波拿巴" (Chinese), "ナポレオン・ボナパルト" (Japanese)
4. Historical event descriptions must use locally accepted expressions
   - Example: "World War II" → "第二次世界大战" (Chinese), "Zweiter Weltkrieg" (German)
5. Preserve ALL original fields (year, type, sources, neutrality_score, etc.) — do NOT modify
6. SEO title in target language should match local search habits
7. Maintain neutral, academic tone — NO emotional additions

# Output Format
Return the SAME JSON structure with `title`, `description`, `local_angle` translated to {target_lang_name}.
Add a `locales.{target_lang_code}` field with translations.

Output JSON only, no markdown wrapping:
{{
  "locales": {{
    "{target_lang_code}": {{
      "title": "<translated>",
      "description": "<translated>",
      "local_angle": "<translated>"
    }}
  }}
}}
"""


def translate_event(event: dict, target_lang_code: str, target_lang_name: str) -> dict:
    """翻译单个事件"""
    prompt = build_translation_prompt(event, target_lang_code, target_lang_name)
    response = client.chat.completions.create(
        model=MODEL,
        messages=[{"role": "user", "content": prompt}],
        temperature=0.2,
        response_format={"type": "json_object"}
    )
    text = response.choices[0].message.content.strip()
    return json.loads(text)


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--input", required=True)
    args = parser.parse_args()

    input_file = Path(args.input)
    data = json.loads(input_file.read_text())
    log.info("Loaded regionalized data from %s", input_file)

    # 对每个地区 × 每个事件 × 每个目标语种进行翻译
    for country_code, country_data in data["regionalized"].items():
        events = country_data.get("events", [])
        log.info("Translating %s (%d events) to %d languages",
                 country_code, len(events), len(TARGET_LANGS))

        for event in events:
            if "locales" not in event:
                event["locales"] = {}

            for lang_code, lang_name in TARGET_LANGS.items():
                if lang_code == "en":
                    # 英语直接复制原字段
                    event["locales"]["en"] = {
                        "title": event["title"],
                        "description": event["description"],
                        "local_angle": event.get("local_angle", "")
                    }
                    continue

                if lang_code in event["locales"]:
                    continue  # 跳过已翻译

                try:
                    result = translate_event(event, lang_code, lang_name)
                    event["locales"][lang_code] = result["locales"][lang_code]
                    log.debug("  ✓ %s/%s/%s", country_code, event["id"][:20], lang_code)
                except Exception as e:
                    log.error("  ✗ %s/%s/%s: %s", country_code, event["id"][:20], lang_code, e)
                    event["locales"][lang_code] = {
                        "title": event["title"],
                        "description": event["description"],
                        "local_angle": event.get("local_angle", ""),
                        "translation_error": str(e)
                    }

    output_file = input_file.parent / f"{input_file.stem}_translated.json"
    output_file.write_text(json.dumps(data, indent=2, ensure_ascii=False))
    log.info("Saved: %s", output_file)


if __name__ == "__main__":
    main()

"""

import os

import requests
import google.generativeai as genai
from dotenv import load_dotenv

load_dotenv()

logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s")
log = logging.getLogger("illustrate")

WIKI_USER_AGENT = os.getenv("WIKI_USER_AGENT", "TodayInHistoryArchive/1.0 (contact@example.com)")
HEADERS = {"User-Agent": WIKI_USER_AGENT}

# Gemini 配置
genai.configure(api_key=os.getenv("GEMINI_API_KEY"))
gemini_model = genai.GenerativeModel("gemini-1.5-flash")


def search_wikimedia_commons(query: str, limit: int = 3) -> list:
    """检索 Wikimedia Commons 公共领域图片"""
    url = "https://commons.wikimedia.org/w/api.php"
    params = {
        "action": "query",
        "format": "json",
        "generator": "search",
        "gsrsearch": f"{query} filetype:bitmap",
        "gsrlimit": limit,
        "prop": "imageinfo",
        "iiprop": "url|extmetadata|mime|size",
        "iiurlwidth": 800
    }

    try:
        resp = requests.get(url, params=params, headers=HEADERS, timeout=20)
        resp.raise_for_status()
        data = resp.json()
        pages = data.get("query", {}).get("pages", {})
        results = []
        for page in pages.values():
            image_info = page.get("imageinfo", [{}])[0]
            ext_meta = image_info.get("extmetadata", {})
            license_short = ext_meta.get("LicenseShortName", {}).get("value", "")

            # 仅接受公共领域或 CC BY-SA
            if license_short not in ("Public domain", "CC BY-SA 4.0", "CC BY-SA 3.0",
                                     "CC BY 4.0", "CC BY 3.0", "CC BY-SA 2.5"):
                continue

            results.append({
                "url": image_info.get("thumburl") or image_info.get("url"),
                "alt": page.get("title", "").replace("File:", "").rsplit(".", 1)[0],
                "caption": ext_meta.get("ImageDescription", {}).get("value", "")[:200],
                "source": "Wikimedia Commons",
                "license": "public-domain" if license_short == "Public domain" else "cc-by-sa",
                "credit": ext_meta.get("Artist", {}).get("value", "")[:100]
            })
        return results
    except Exception as e:
        log.error("Wikimedia search failed for '%s': %s", query, e)
        return []


def generate_with_gemini(event: dict) -> dict:
    """使用 Gemini 生成历史场景插画（备选方案）"""

- Aspect ratio: 16:9 (landscape)
- Color palette: Muted earth tones, period-appropriate

Return the image directly."""

    try:
        response = gemini_model.generate_content(prompt)
        # Gemini 1.5 Flash image generation（如支持）
        # 注意：实际使用时需调用支持图片生成的 Gemini 版本
        # 此处简化为返回占位信息
        return {
            "url": None,  # 实际应保存到 R2 并返回 URL
            "alt": f"AI-generated illustration: {event['title']}",
            "caption": f"AI-generated illustration of {event['year']} event",
            "source": "Gemini 1.5 Flash",
            "license": "ai-generated",
            "credit": "Today in History Archive"
        }
    except Exception as e:
        log.error("Gemini generation failed: %s", e)
        return None


def illustrate_event(event: dict) -> dict:
    """为单个事件配图"""
    # 1. 先尝试 Wikimedia Commons
    query = event["title"]
    images = search_wikimedia_commons(query, limit=3)
    if images:
        log.info("  ✓ Wikimedia: %s", event["title"][:40])
        return images[0]

    # 2. 备选：Gemini 生成
    log.info("  → No Wikimedia result, trying Gemini: %s", event["title"][:40])
    gemini_image = generate_with_gemini(event)
    if gemini_image:
        return gemini_image

    # 3. 兜底：无配图
    return None

  - **L3 敏感事件强制人工**：sensitivity_flags 非空

### 6.2 完整 `audit.py` 脚本

```python
#!/usr/bin/env python3
"""
内容审核 — GPT-4o 多源事实核查 + 敏感事件检测 + 中立性评估
依赖：openai, python-dotenv
运行：python3 scripts/audit.py --input data/events/07-04_illustrated.json
"""
import os
import sys
import json
import logging
import argparse
from pathlib import Path
from openai import OpenAI
from dotenv import load_dotenv

load_dotenv()

logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s")
log = logging.getLogger("audit")

client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))
MODEL = "gpt-4o"

# 敏感事件关键词（多语种）
SENSITIVE_KEYWORDS = {
    "war": ["war", "invasion", "battle", "siege", "occupation", "war crime", "战", "侵", "战争"],
    "genocide": ["genocide", "holocaust", "massacre", "ethnic cleansing", "屠杀", "种族灭绝"],
    "political_repression": ["purge", "execution", "political prisoner", "dissident", "清洗", "镇压"],
    "religious": ["religious conflict", "sectarian", "crusade", "jihad", "宗教冲突"],
    "territorial_dispute": ["annexation", "secession", "border dispute", "吞并", "独立"],
    "terrorism": ["terrorist", "bombing", "attack", "爆炸", "袭击"]
}


def detect_sensitive_keywords(text: str) -> list:
    """检测敏感关键词"""
    text_lower = text.lower()
    flags = []
    for flag, keywords in SENSITIVE_KEYWORDS.items():
        if any(kw in text_lower for kw in keywords):
            flags.append(flag)
    return flags or ["none"]


def build_audit_prompt(event: dict, country_code: str) -> str:
    """构建审核 Prompt"""
    return f"""You are a fact-checker and neutrality reviewer for a historical archive.

# Task
Audit the following historical event for {country_code}:

```json
{json.dumps(event, ensure_ascii=False, indent=2)}
```

# Audit Dimensions

## 1. Fact Check
- Verify year, key figures, places against the source URLs
- Flag any factual inaccuracies
- Confirm at least 2 independent sources

## 2. Neutrality Assessment (0.0-1.0)
- 1.0: Perfectly neutral, multiple perspectives
- 0.8: Mostly neutral, minor bias
- 0.5: Some biased language
- 0.3: Clear bias, needs revision
- 0.0: Pure propaganda

## 3. Sensitive Topic Detection
Check for:
- War/military events: multiple perspectives?
- Political repression: government + victim views?
- Religious conflicts: balanced treatment?
- Territorial disputes: neutral terminology?
- Terrorism: factual description, no glorification

## 4. Source Quality
- Are sources authoritative (Wikipedia, national archives)?
- Are sources diverse (not all from one country)?
- Are sources recent enough?

## 5. Sensitive Country-Specific Rules
- Germany (de): Nazi references must use academic terminology, no glorification
- China (cn): Use official terms where required, present multiple perspectives
- Turkey (tr): Armenian events must be factual, present international scholarship
- Russia (ru) / Ukraine (ua): Use neutral terminology, present both perspectives
- Japan (jp): War crimes references must be factual, acknowledge mainstream historiography

# Output Format (JSON)
{{
  "audit_id": "<event_id>",
  "country_code": "{country_code}",
  "fact_check": {{
    "score": <0.0-1.0>,
    "issues": ["<issue 1>", "<issue 2>"],
    "verified_facts": ["<verified fact 1>"]
  }},
  "neutrality": {{
    "score": <0.0-1.0>,
    "bias_detected": ["<bias description>"],
    "suggestions": ["<fix suggestion>"]
  }},
  "sensitivity": {{
    "flags": ["<flag>", ...],
    "requires_human_review": <true|false>,
    "reason": "<why human review needed>"
  }},
  "sources": {{
    "quality_score": <0.0-1.0>,
    "diversity": <true|false>,
    "missing_perspectives": ["<perspective>"]
  }},
  "overall_verdict": "pass|revise|reject|human_review",
  "summary": "<1-2 sentence summary>"
}}

Return JSON only, no markdown.
"""


def audit_event(event: dict, country_code: str) -> dict:
    """审核单个事件"""
    prompt = build_audit_prompt(event, country_code)
    response = client.chat.completions.create(
        model=MODEL,
        messages=[{"role": "user", "content": prompt}],
        temperature=0.1,  # 极低温度保证审核一致性
        response_format={"type": "json_object"}
    )
    text = response.choices[0].message.content.strip()
    return json.loads(text)


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--input", required=True)
    args = parser.parse_args()

    input_file = Path(args.input)
    data = json.loads(input_file.read_text())

    audit_report = {
        "date": data["date"],
        "audited_at": __import__("datetime").datetime.utcnow().isoformat() + "Z",
        "model": MODEL,
        "results": {},
        "summary": {"pass": 0, "revise": 0, "reject": 0, "human_review": 0}
    }

    for country_code, country_data in data["regionalized"].items():
        events = country_data.get("events", [])
        country_results = []

        for event in events:
            # 1. 关键词预检
            event["sensitivity_flags"] = detect_sensitive_keywords(
                event.get("title", "") + " " + event.get("description", "")
            )

            # 2. AI 审核
            try:
                audit = audit_event(event, country_code)
                country_results.append(audit)

                # 更新事件中立性评分
                event["neutrality_score"] = audit["neutrality"]["score"]

                # 统计
                verdict = audit["overall_verdict"]
                audit_report["summary"][verdict] = audit_report["summary"].get(verdict, 0) + 1

                log.info("  %s [%s] %s: %.2f",
                         country_code, verdict, event["title"][:30],
                         audit["neutrality"]["score"])

            except Exception as e:
                log.error("Audit failed for %s/%s: %s", country_code, event["id"], e)
                country_results.append({
                    "audit_id": event["id"],
                    "country_code": country_code,
                    "overall_verdict": "human_review",
                    "error": str(e)
                })
                audit_report["summary"]["human_review"] += 1

        audit_report["results"][country_code] = country_results

    # 保存审核报告
    audit_file = input_file.parent / f"{input_file.stem}_audit.json"
    audit_file.write_text(json.dumps(audit_report, indent=2, ensure_ascii=False))
    log.info("Audit report saved: %s", audit_file)
    log.info("Summary: %s", audit_report["summary"])

    # 标记事件审核状态
    for country_code, results in audit_report["results"].items():
        for event, result in zip(data["regionalized"][country_code]["events"], results):
            event["_audit"] = {
                "verdict": result.get("overall_verdict"),
                "neutrality": result.get("neutrality", {}).get("score"),
                "requires_human": result.get("sensitivity", {}).get("requires_human_review", False)
            }

    output_file = input_file.parent / f"{input_file.stem}_audited.json"
    output_file.write_text(json.dumps(data, indent=2, ensure_ascii=False))
    log.info("Audited data saved: %s", output_file)


if __name__ == "__main__":
    main()
```

### 6.3 审核决策矩阵

| AI 审核结果 | 中立性评分 | 敏感性 | 处理 |
|------------|----------|--------|------|
| pass | ≥ 0.8 | none | 自动发布 |
| pass | ≥ 0.8 | war/religious | 自动发布 + 标记观察 |
| revise | 0.5-0.7 | any | AI 自动修订 1 次，再审 |
| reject | < 0.5 | any | 不发布，归档错误案例 |
| human_review | any | any | 强制人工审核 |

### 6.4 人工抽检 SOP

**抽检范围：**
- 5% 主要国家（美国/英国/德国/中国/日本/俄罗斯）
- 100% 敏感事件（sensitivity_flags 非空）
- 100% AI 标记 human_review 的事件

**抽检流程：**
1. 编辑打开审核报告 `_audit.json`
2. 按事件 ID 在源数据中查找
3. 检查 4 项：事实准确性 / 中立性 / 引用完整性 / 敏感性
4. 决策：通过 / 修订 / 拒绝
5. 记录决策到 `audit_log_YYYYMMDD.md`

### 7.2 完整 `publish.py` 脚本

```python
#!/usr/bin/env python3
"""
内容发布 — 将审核通过的内容生成为 Hugo Markdown 并提交 Git
依赖：无（标准库）
运行：python3 scripts/publish.py --input data/events/07-04_audited.json
"""
import os
import sys
import json
import logging
import argparse
import subprocess
from datetime import datetime
from pathlib import Path

logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s")
log = logging.getLogger("publish")

REPO_ROOT = Path(__file__).parent.parent
CONTENT_DIR = REPO_ROOT / "content"
DATE_FORMATS = {
    "en": "January 2, 2006",
    "zh": "2006年1月2日",
    "ja": "2006年1月2日",
    "ko": "2006년 1월 2일",
    "ru": "2 января 2006 г.",
    "default": "2 January 2006",
}

LANGUAGES = ["en", "zh", "es", "fr", "de", "it", "pt", "ru", "ja", "ko"]

COUNTRY_NAMES = {
    "us": {"en": "United States", "zh": "美国", "es": "Estados Unidos", "fr": "États-Unis",
           "de": "Vereinigte Staaten", "it": "Stati Uniti", "pt": "Estados Unidos",
           "ru": "США", "ja": "アメリカ合衆国", "ko": "미국"},
    # ... 其余 29 地区元数据，完整版从 data/countries.json 加载
}


def load_country_names():
    """从 data/countries.json 加载完整地区元数据"""
    countries_file = REPO_ROOT / "data" / "countries.json"
    if countries_file.exists():
        data = json.loads(countries_file.read_text())
        return {c["code"]: c["local_names"] for c in data["countries"]}
    return COUNTRY_NAMES


def month_day_pretty(month: int, day: int, lang: str) -> str:
    """格式化月日为本地化字符串"""
    formats = {
        "en": f"{datetime(2000, month, day).strftime('%B')} {day}",
        "zh": f"{month}月{day}日",
        "ja": f"{month}月{day}日",
        "ko": f"{month}월 {day}일",
        "ru": f"{day} {datetime(2000, month, day).strftime('%B')}",
        "default": f"{day} {datetime(2000, month, day).strftime('%B')}",
    }
    return formats.get(lang, formats["default"])


def generate_markdown(event: dict, lang: str) -> str:
    """生成单个事件的 Markdown 段落"""
    locale = event.get("locales", {}).get(lang, event.get("locales", {}).get("en", {}))
    title = locale.get("title", event["title"])
    description = locale.get("description", event["description"])

    md = f"## {event['year']} — {title}\n\n"
    md += f"{description}\n\n"

    if event.get("image"):
        img = event["image"]
        md += f"![{img.get('alt', '')}]({img['url']})\n\n"
        md += f"*{img.get('caption', '')} — Source: {img.get('source', '')} ({img.get('license', '')})*\n\n"

    if event.get("sources"):
        md += "**Sources:**\n"
        for src in event["sources"]:
            md += f"- [{src['name']}]({src['url']})\n"
        md += "\n"

    return md


def generate_page(country_code: str, country_name: str, country_data: dict,
                  month: int, day: int, lang: str) -> str:
    """生成单页 Markdown 内容"""
    events = [e for e in country_data.get("events", [])
              if e.get("_audit", {}).get("verdict") in ("pass", None)]

    if not events:
        return None  # 无可发布事件

    pretty_date = month_day_pretty(month, day, lang)

    # Front Matter
    front_matter = f"""---
title: "{pretty_date} — {country_name}"
date: 2026-{month:02d}-{day:02d}
country_code: "{country_code}"
country_name: "{country_name}"
month_day: "{month:02d}-{day:02d}"
month_day_iso: "{month:02d}-{day:02d}"
month_day_pretty: "{pretty_date}"
events_count: {len(events)}
date_format: "{DATE_FORMATS.get(lang, DATE_FORMATS['default'])}"
description: "On this day in {country_name} history: {len(events)} events documented for {pretty_date}."
og_image: "/images/og-{country_code}.png"
draft: false
---

"""

    # 正文
    body = f"# On This Day in {country_name} — {pretty_date}\n\n"
    body += f"This page documents {len(events)} historical events that occurred on {pretty_date} "
    body += f"with relevance to {country_name}.\n\n"

    # 事件列表
    for event in events:
        body += generate_markdown(event, lang)

    # AI 披露
    body += "\n---\n\n"
    body += "*This content was AI-assisted and reviewed by human editors. "
    body += "Sources are cited for each event. [Report an error](/contact/).*\n"

    # 关联推荐（同日其他国家）
    body += "\n## Related: On This Day in Other Countries\n\n"
    body += f"- [On This Day — All Countries]({f'/{lang}/on-this-day/all/{month:02d}-{day:02d}/'})\n"

    return front_matter + body


def publish_for_language(data: dict, lang: str, country_names: dict):
    """为单个语种生成所有地区页面"""
    month = data["date"]["month"]
    day = data["date"]["day"]

    for country_code, country_data in data["regionalized"].items():
        country_name = country_names.get(country_code, {}).get(lang, country_code.upper())
        md = generate_page(country_code, country_name, country_data, month, day, lang)
        if md is None:
            continue

        output_dir = CONTENT_DIR / lang / country_code
        output_dir.mkdir(parents=True, exist_ok=True)
        output_file = output_dir / f"{month:02d}-{day:02d}.md"
        output_file.write_text(md)
        log.info("✓ %s/%s/%02d-%02d.md", lang, country_code, month, day)


def git_commit_and_push(date_str: str):
    """提交并推送新内容"""
    try:
        subprocess.run(["git", "add", "content/", "data/"], check=True, cwd=REPO_ROOT)
        result = subprocess.run(
            ["git", "diff", "--staged", "--quiet"],
            cwd=REPO_ROOT
        )
        if result.returncode == 0:
            log.info("No changes to commit")
            return

        subprocess.run(
            ["git", "commit", "-m", f"chore(content): daily update {date_str}"],
            check=True, cwd=REPO_ROOT
        )
        subprocess.run(["git", "push"], check=True, cwd=REPO_ROOT)
        log.info("Git push successful")
    except subprocess.CalledProcessError as e:
        log.error("Git operation failed: %s", e)


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--input", required=True)
    parser.add_argument("--no-commit", action="store_true", help="Skip git commit")
    args = parser.parse_args()

    input_file = Path(args.input)
    data = json.loads(input_file.read_text())
    log.info("Publishing %s-%02d-%02d", data["date"].get("year", 2026),
             data["date"]["month"], data["date"]["day"])

    country_names = load_country_names()

    for lang in LANGUAGES:
        publish_for_language(data, lang, country_names)

    if not args.no_commit:
        date_str = f"{data['date']['month']:02d}-{data['date']['day']:02d}"
        git_commit_and_push(date_str)


if __name__ == "__main__":
    main()
```

### 12.2 质量校验工具

**`scripts/validate_content.py`：**

```python
#!/usr/bin/env python3
"""内容质量校验工具 — 每日发布后运行"""
import json
import sys
from pathlib import Path
from jsonschema import validate

def check_event_schema(event):
    """校验事件 schema"""
    required = ["id", "year", "title", "description", "type", "countries", "sources", "neutrality_score"]
    for field in required:
        if field not in event:
            return False, f"Missing field: {field}"
    if event["neutrality_score"] < 0.7:
        return False, f"Neutrality {event['neutrality_score']} below 0.7"
    if len(event["sources"]) < 2:
        return False, f"Only {len(event['sources'])} sources (need 2)"
    return True, "OK"

def check_page_integrity(md_file):
    """校验页面完整性"""
    content = md_file.read_text()
    if "AI-assisted" not in content:
        return False, "Missing AI disclosure"
    if "Sources:" not in content:
        return False, "Missing sources section"
    if len(content) < 500:
        return False, f"Content too short: {len(content)} chars"
    return True, "OK"

def main():
    content_dir = Path("content")
    errors = []
    total = 0

    for md_file in content_dir.rglob("*.md"):
        if md_file.name == "_index.md":
            continue
        total += 1
        ok, msg = check_page_integrity(md_file)
        if not ok:
            errors.append(f"{md_file}: {msg}")

    print(f"Checked {total} pages, {len(errors)} errors")
    for err in errors[:10]:
        print(f"  ✗ {err}")

    sys.exit(1 if errors else 0)

if __name__ == "__main__":
    main()
```
